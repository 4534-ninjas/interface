Protocol BNF:

packet := <header> <payload>

header := '\xff' (checksum?)
payload := <sensor_msg> | <sensor_def> | <debug_msg> | <debug_def> | <ctl_msg>

sensor_def := <int32 min> <bool enabled> <int32 max> <str units> <str description>
sensor_msg := <int32 val>

debug_def := <int32 id> <bool enabled> <str source> <str fmt_string>
debug_msg := <int32 id> <int32 counter> <str msg>
